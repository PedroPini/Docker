Comandos
#docker run cria um container com a imagem passada como parametro
docker run nome_da_imagem


#traz a versão intalada do docker 
docker version - versao do docker


#lista todos os containers ATIVOS no momento
docker ps  


#lista todos os conteiners inclusive os parados
docker ps -a 


#executa algo dentro do container
docker run ubuntu echo "Ola Mundo"


#associa o terminal com o container fazendo um terminal interativo
docker run -it nome_da_imagem


#sair do container interativo
exit


#Startar container
docker start id_do_container


#Parar Container
docker stop  id_do_container


#Remover container
docker rm  id_do_container


#Remover Containers parados(inativos)
docker container prune


#Remover Imagem
docker rmi  nome_do_repositorio


#Rodar imagem em background
docker run -d nome_da_imagem


#Retorna as portas que estão sendo utilizadas pelo container
docker port  id_do_container


#Colocar apelido(alias) no container
docker run --name nome_escolhido nome_da_imagem


#Listar o id de todos os containers ativos rodando
docker ps -q


#Parar multiplos containers ativos sem esperar o tempo default de 10 segundos
docker stop -t 0 $(docker ps -q)
